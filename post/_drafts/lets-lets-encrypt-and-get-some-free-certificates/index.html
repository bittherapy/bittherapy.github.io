Let’s Let’s Encrypt and get some free certificates

apt-get install letsencrypt -t jessie-backports

certbot certonly --webroot -w /var/www/letsencrypt -d bittherapy.net -d www.bittherapy.net

certbot renew --dry-run

/etc/letsencrypt/live/bittherapy.net/fullchain.pem
/etc/letsencrypt/live/bittherapy.net/privkey.pem

ssl_certificate /etc/letsencrypt/live/bittherapy.net/fullchain.pem;  
ssl_certificate_key /etc/letsencrypt/live/bittherapy.net/privkey.pem;

location ~ .well-known/acme-challenge/ {
    root /var/www/letsencrypt;
    default_type text/plain;
}
